020    package org.crsh.util;
021    
022    import java.io.Closeable;
023    import java.io.Flushable;
024    import java.io.IOException;
025    import java.io.Writer;
026    
027    /**
028     * @author <a href="mailto:julien.viet@exoplatform.com">Julien Viet</a>
029     * @version $Revision$
030     */
031    public class AppendableWriter extends Writer {
032    
033      /** . */
034      private final Appendable out;
035    
036      /** . */
037      private boolean closed;
038    
039      public AppendableWriter(Appendable out) {
040        this.out = out;
041      }
042    
043      @Override
044      public void write(char[] cbuf, int off, int len) throws IOException {
045        if (closed) {
046          throw new IOException("Already closed");
047        }
048        int end = off + len;
049        while (off < end) {
050          out.append(cbuf[off++]);
051        }
052      }
053    
054      @Override
055      public void flush() throws IOException {
056        if (closed) {
057          throw new IOException("Already closed");
058        }
059        if (out instanceof Flushable) {
060          ((Flushable)out).flush();
061        }
062      }
063    
064      @Override
065      public void close() throws IOException {
066        if (!closed) {
067          closed = true;
068          if (out instanceof Closeable) {
069            ((Closeable)out).close();
070          }
071        }
072      }
073    }
