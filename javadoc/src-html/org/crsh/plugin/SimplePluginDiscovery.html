019    package org.crsh.plugin;
020    
021    import java.util.Collections;
022    import java.util.LinkedHashSet;
023    
024    /**
025     * A simple implemention of {@link PluginDiscovery} where the plugins are explictly provided.
026     * 
027     * @author <a href="mailto:julien.viet@exoplatform.com">Julien Viet</a>
028     */
029    public class SimplePluginDiscovery implements PluginDiscovery {
030    
031      /** . */
032      private final LinkedHashSet<CRaSHPlugin<?>> plugins;
033    
034      public SimplePluginDiscovery() {
035        this.plugins = new LinkedHashSet<CRaSHPlugin<?>>();
036      }
037    
038      public SimplePluginDiscovery(CRaSHPlugin<?>... plugins) {
039        this();
040        for (CRaSHPlugin<?> plugin : plugins) {
041          add(plugin);
042        }
043      }
044      /**
045       * Add a plugin.
046       *
047       * @param plugin the plugin
048       * @return this object
049       * @throws NullPointerException if the plugin is null
050       */
051      public SimplePluginDiscovery add(CRaSHPlugin<?> plugin) throws NullPointerException {
052        if (plugin == null) {
053          throw new NullPointerException();
054        }
055        plugins.add(plugin);
056        return this;
057      }
058    
059      public Iterable<CRaSHPlugin<?>> getPlugins() {
060        return Collections.unmodifiableSet(plugins);
061      }
062    }
