020    package org.crsh.cmdline.completers;
021    
022    import org.crsh.cmdline.ParameterDescriptor;
023    import org.crsh.cmdline.spi.Completer;
024    import org.crsh.cmdline.spi.ValueCompletion;
025    
026    /**
027     * A {@link Completer} implementation that returns no completion results.
028     *
029     * @author <a href="mailto:julien.viet@exoplatform.com">Julien Viet</a>
030     * @version $Revision$
031     */
032    public class EmptyCompleter implements Completer {
033    
034      /** . */
035      private static final EmptyCompleter instance = new EmptyCompleter();
036    
037      /**
038       * Returns the empty completer instance.
039       *
040       * @return the instance
041       */
042      public static EmptyCompleter getInstance() {
043        return instance;
044      }
045    
046      /**
047       * Returns the value returned by {@link java.util.Collections#emptyList()}.
048       */
049      public ValueCompletion complete(ParameterDescriptor<?> parameter, String prefix) {
050        return ValueCompletion.create();
051      }
052    }
