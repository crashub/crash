020    package org.crsh.jcr;
021    
022    import org.apache.sshd.server.Command;
023    import org.crsh.cmdline.ClassDescriptor;
024    import org.crsh.cmdline.CommandFactory;
025    import org.crsh.cmdline.IntrospectionException;
026    import org.crsh.cmdline.matcher.CmdLineException;
027    import org.crsh.cmdline.matcher.CommandMatch;
028    import org.crsh.cmdline.matcher.InvocationContext;
029    import org.crsh.cmdline.matcher.Matcher;
030    import org.crsh.ssh.term.FailCommand;
031    import org.crsh.ssh.term.scp.CommandPlugin;
032    import org.crsh.ssh.term.scp.SCPAction;
033    
034    /**
035     * @author <a href="mailto:julien.viet@exoplatform.com">Julien Viet</a>
036     * @version $Revision$
037     */
038    public class SCPCommandPlugin extends CommandPlugin {
039    
040      @Override
041      public Command createCommand(String command) {
042        //
043        if (command.startsWith("scp ")) {
044          try {
045            command = command.substring(4);
046            SCPAction action = new SCPAction();
047            ClassDescriptor<SCPAction> descriptor = CommandFactory.create(SCPAction.class);
048            Matcher<SCPAction> analyzer = Matcher.createMatcher("main", descriptor);
049            CommandMatch<SCPAction, ?, ?> match = analyzer.match(command);
050            match.invoke(new InvocationContext(), action);
051            if (Boolean.TRUE.equals(action.isSource())) {
052              return new SourceCommand(action.getTarget(), Boolean.TRUE.equals(action.isRecursive()));
053            }
054            else if (Boolean.TRUE.equals(action.isSink())) {
055              return new SinkCommand(action.getTarget(), Boolean.TRUE.equals(action.isRecursive()));
056            }
057            else {
058              return new FailCommand("Cannot execute command " + command);
059            }
060          }
061          catch (CmdLineException e) {
062            return new FailCommand("Cannot execute command " + command, e);
063          }
064          catch (IntrospectionException e) {
065            return new FailCommand("Cannot execute command " + command, e);
066          }
067        } else {
068          return new FailCommand("Cannot execute command " + command);
069        }
070      }
071    }
