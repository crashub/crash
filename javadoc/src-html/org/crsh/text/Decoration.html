001    package org.crsh.text;
002    
003    import java.io.Serializable;
004    
005    /**
006     * @author <a href="mailto:alain.defrance@exoplatform.com">Alain Defrance</a>
007     */
008    public enum Decoration implements Serializable {
009    
010      bold(1),
011      underline(4),
012      blink(5);
013    
014      /** . */
015      public final int code;
016    
017      /** . */
018      public final Style style;
019    
020      private Decoration(int code) {
021        this.code = code;
022        this.style = Style.style(this);
023      }
024    }
